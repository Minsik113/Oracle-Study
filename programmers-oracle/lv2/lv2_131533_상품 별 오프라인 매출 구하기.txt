-- OFFLINE_SALE PRODUCT 
SELECT PRODUCT_CODE, PRICE*TOTAL SALES
FROM (SELECT PRODUCT_CODE, PRICE, SUM(B.SALES_AMOUNT) TOTAL
    FROM PRODUCT A, OFFLINE_SALE B
    WHERE A.PRODUCT_ID = B.PRODUCT_ID
    GROUP BY A.PRODUCT_CODE,A.PRICE)
ORDER BY SALES DESC, PRODUCT_CODE ASC
;
